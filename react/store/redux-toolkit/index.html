Redux Toolkit ​

Redux Toolkit 是官方推荐的编写 Redux 逻辑的方法，用以代替传统的 Redux 独立使用。 Redux Toolkit 构建了 Redux 建议的最佳实践，简化了大多数 Redux 任务，防止了常见错误，并使编写 Redux 应用程序变得更加容易。

安装 ​

sh
pnpm add @reduxjs/toolkit

如果需要在 React 应用程序中使用还需要补充安装 react-redux：

jsx
pnpm add react-redux

使用 ​

定义状态切片 ​

src/store/counter.js

jsx
// 状态切片（就是模块）
// 也是提供数据和修改数据的方法
import { createSlice, createAsyncThunk } from "@reduxjs/toolkit";

const counterSlice = createSlice({
  // 切片名称，命名空间
  name: "counter",
  // 定义切片的初始数据
  initialState: {
    count: 1,
  },
  // 定义修改状态的同步方法（这里定义的都是同步任务）
  reducers: {
    // 定义的方法可以通过参数获取当前切片的最新状态
    add(state) {
      // state 其实是当前切片状态的代理对象
      // 想要修改状态可以直接在这个对象上进行属性修改，不需要定义返回值
      state.count++;
    },
    sub(state, action) {
      // action 触发当前任务的描述对象

      let { payload } = action;
      state.count -= payload;
    },
  },
});

// 导出切片名称，用于为 store 定义命名空间
export const counterSliceName = counterSlice.name;
// 导出切片的状态修改函数，为 store 合并全部的初始数据，以及可以执行的任务
export const counterSliceReducer = counterSlice.reducer;
// 导出切片提供的 action 对象创建函数
export const { add, sub } = counterSlice.actions;

// 创建一个包含有异步任务的 action 创建函数
// 参数一：action 对应的 type
// 参数二：要求有返回异步结果的函数，返回值可以是普通数据，也可以是 Promise 对象
export const subByAsync = createAsyncThunk(
  `${counterSliceName}/subByAsync`,
  (payload, { getState, dispatch }) => {
    // 获取组件给定的初始值
    console.log("payload", payload);

    // 执行封装的异步任务
    ajax(payload).then((res) => {
      console.log("res", res);

      // 异步任务结束后，向仓库派发任务，结合异步任务结果和当前状态，继续修改切片状态
      let state = getState();
      console.log("state", state);

      // 可以选择调用同步任务，或者继续派发异步任务
      dispatch(sub(res));
    });

    // return 100;
  }
);

// 简化
export const subByAsync2 = createAsyncThunk(
  `${counterSliceName}/subByAsync2`,
  async (payload) => {
    let res = await ajax(payload);
    // 可以将异步任务结果返回到组件内
    return res;
  }
);

/**
 * 模仿异步
 * @param {*} data
 * @returns
 */
async function ajax(data) {
  return data * 2;
}

创建全局状态库 ​

src/store/index.js

jsx
import { configureStore } from "@reduxjs/toolkit";

import { counterSliceName, counterSliceReducer } from "./counter.slice";

// 定义全局状态存储对象
const store = configureStore({
  reducer: {
    // 为仓库添加命名空间以及对应的 reducer
    [counterSliceName]: counterSliceReducer,
  },
});

export default store;

注入 ​

src/main.jsx

jsx
import React from "react";
import ReactDOM from "react-dom/client";
import { Provider } from "react-redux";
import store from "./store";
import App from "./App.jsx";

ReactDOM.createRoot(document.getElementById("root")).render(
  <Provider store={store}>
  	<App />
  </Provider>
);

src/App.jsx

jsx
import { Provider } from "react-redux";
import store from "./store";
import Counter from "./views/counter.jsx";

function App() {
  // 利用 Provider 组件将仓库注入到 react 所有组件内
  return (
    <Provider store={store}>
      {/* 路由 */}
    </Provider>
  );
}

export default App;

使用 ​

src/view/counter.jsx

jsx
import { useSelector, useDispatch } from "react-redux";
import { add, sub, subByAsync, subByAsync2 } from "../store/counter.slice";

/* function useFn() {
  const dispatch = useDispatch();
  dispatch({ type: "counter/add" });
} */

const Counter = () => {
  // 但是，封装的函数在函数组件内执行是可以的
  // 允许将多个函数组件的公共逻辑进行抽离和封装
  // useFn();
  // hook 函数只能作为函数组件内的最外层函数执行，不能放在条件或者循环语句内
  // hook 的原理是保证函数组件在刷新时一定会同步执行的函数

  // 从仓库取出需要的状态
  const counter = useSelector((state) => {
    // 回调函数的参数是仓库所有状态，返回值是组件需要的状态
    console.log("仓库状态", state);
    return state.counter;
  });

  const getAsyncResponse = () => {
    dispatch(subByAsync2(300)).then((res) => {
      console.log("组件内获取异步任务返回值", res.payload);

      // 继续完成其它逻辑
      dispatch(sub(res.payload));
    });
  };

  const dispatch = useDispatch();

  return (
    <fieldset>
      <legend>计数器</legend>
      <p>{counter.count}</p>
      {/* 执行 dispatch 方法参数需要传递一个 action 对象
      对象需要 type 和 payload 属性
      其中，type 要求补充切片的命名空间 */}
      <button onClick={() => dispatch({ type: "counter/add" })}>增加</button>
      {/* 可以直接使用切片提供的 action 创建函数，快速生成一个对象 */}
      <button onClick={() => dispatch(add())}>增加</button>
      {/* 任务的 payload 可以通过函数参数传递 */}
      <button onClick={() => dispatch(sub(10))}>减少</button>
      <button onClick={() => dispatch(subByAsync(100))}>异步减少</button>
      <button onClick={() => getAsyncResponse()}>异步减少2</button>
    </fieldset>
  );
};

export default Counter;
